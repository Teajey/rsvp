package rsvp // import "github.com/Teajey/rsvp"


CONSTANTS

const (
	SupportedMediaTypePlaintext string = "text/plain"
	SupportedMediaTypeHtml      string = "text/html"
	SupportedMediaTypeBytes     string = "application/octet-stream"
	SupportedMediaTypeJson      string = "application/json"
	SupportedMediaTypeXml       string = "application/xml"
	SupportedMediaTypeGob       string = "application/vnd.golang.gob"
)

VARIABLES

var AllSupportedTypes = []string{
	SupportedMediaTypePlaintext,
	SupportedMediaTypeHtml,
	SupportedMediaTypeBytes,
	SupportedMediaTypeJson,
	SupportedMediaTypeXml,
	SupportedMediaTypeGob,
}
    Can be used to place every supported type on:
      - Config.RenderSeeOtherBlackList
      - Config.RenderFoundBlackList


TYPES

type Config struct {
	HtmlTemplate *html.Template
	TextTemplate *text.Template
	RenderPermanentRedirect bool
	RenderMovedPermanently bool
	RenderSeeOtherBlackList []string
	RenderFoundBlackList []string
	// Has unexported fields.
}
    Settings for writing the rsvp.Response

func DefaultConfig() *Config

type Handler interface {
	ServeHTTP(h http.Header, r *http.Request) Response
}

type HandlerFunc func(h http.Header, r *http.Request) Response

type Response struct {
	Body         any
	TemplateName string
	Status       int
	// Has unexported fields.
}

func Found(url string, body any) Response

func MovedPermanently(url string) Response

func Ok() Response

func PermanentRedirect(url string) Response

func SeeOther(url string, body any) Response

func (r *Response) Html(html string)

func (res *Response) Write(w http.ResponseWriter, r *http.Request, cfg *Config) error

type ServeMux struct {
	Std    *http.ServeMux
	Config *Config
}

func NewServeMux() *ServeMux

func (m *ServeMux) Handle(pattern string, handler Handler)

func (m *ServeMux) HandleFunc(pattern string, handler HandlerFunc)

func (m *ServeMux) MiddleHandleFunc(pattern string, handler func(h http.ResponseWriter, r *http.Request) Response)

func (m *ServeMux) ServeHTTP(w http.ResponseWriter, r *http.Request)
